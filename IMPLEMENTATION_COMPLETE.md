```typescript
const getAllOrders = (conversations: Conversation[]) => {
  const orders: Array<{
    message: Message;
    conversationId: string;
    aiName: string;
  }> = [];
  
  conversations.forEach(conv => {
    conv.messages
      .filter(msg => msg.order)
      .forEach(msg => {
        orders.push({
          message: msg,
          conversationId: conv.id,
          aiName: conv.characterSettings?.nickname || 'AI'
        });
      });
  });
  
  // 按时间倒序
  return orders.sort((a, b) => b.message.timestamp - a.message.timestamp);
};
```
