## 💡 技术实现

### 图片上传
```typescript
const handlePhotoUpload = (file: File, photoNum: 1 | 2) => {
  const reader = new FileReader();
  reader.onloadend = () => {
    const base64 = reader.result as string;
    if (photoNum === 1) {
      setPhoto1(base64);
      localStorage.setItem('homePhoto1', base64);
    }
  };
  reader.readAsDataURL(file);
};
```

### 壁纸显示
```tsx
{wallpaper ? (
  <div className="absolute inset-0">
    <img src={wallpaper} className="w-full h-full object-cover" />
    <div className="absolute inset-0 bg-black/10"></div>
  </div>
) : (
  <div className="bg-gradient-to-br from-pink-100 ...">
    {/* 默认装饰 */}
  </div>
)}
```

### 时间样式
```tsx
<div className="text-7xl font-extralight text-white tracking-tighter drop-shadow-2xl" 
     style={{ textShadow: '0 2px 10px rgba(0,0,0,0.3)' }}>
  {formatTime()}
</div>
```
